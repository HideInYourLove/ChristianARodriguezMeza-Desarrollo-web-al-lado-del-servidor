    <script>
        // Función para calcular el coste total
        function calcularCoste() {
            // Obtener elementos del DOM
            const alojamiento = parseFloat(document.getElementById('alojamiento').value) || 0;
            const alimentacion = parseFloat(document.getElementById('alimentacion').value) || 0;
            const entretenimiento = parseFloat(document.getElementById('entretenimiento').value) || 0;

            // Calcular el total
            const total = alojamiento + alimentacion + entretenimiento;

            // Mostrar resultados usando el DOM
            document.getElementById('resAlojamiento').textContent = alojamiento;
            document.getElementById('resAlimentacion').textContent = alimentacion;
            document.getElementById('resEntretenimiento').textContent = entretenimiento;
            document.getElementById('costoTotal').textContent = '$' + total;

            // Mostrar el div de resultado
            document.getElementById('resultado').style.display = 'block';
        }
    </script>
